```cpp
class Vec3 {
public: // 这里的 public 意味着下面的东西是对外公开的，谁都可以访问
// 1. 成员变量 (这个类肚子里装了什么数据？)
double e[3]; // 我们用一个包含3个数字的数组来存 x, y, z

// 2. 构造函数 (当别人创建这个对象时，怎么初始化它？)
// 注意：构造函数的名字必须和类名一模一样，且没有返回值！
Vec3() {
e[0] = 0.0;
e[1] = 0.0;
e[2] = 0.0;
}
  
// 这是一个带参数的构造函数，方便我们直接赋值
Vec3(double e0, double e1, double e2) {
e[0] = e0;
e[1] = e1;
e[2] = e2;
}
  
// 3. 成员函数 (这个类能提供什么服务？)
// 比如，获取 x 坐标
double x() const { return e[0]; } // 这里的 const 表示这个函数绝对不会修改类里面的数据
double y() const { return e[1]; }
double z() const { return e[2]; }
};

  
inline Vec3 operator+(const Vec3 &u, const Vec3 &v){
return Vec3(u.e[0] + v.e[0], u.e[1] + v.e[1], u.e[2] + v.e[2]);
}

inline Vec3 operator-(const Vec3 &u, const Vec3 &v){
return Vec3(u.e[0] - v.e[0], u.e[1] - v.e[1], u.e[2] - v.e[2]);
}
  
inline Vec3 operator*(double t, const Vec3 &v){
return Vec3(t * v.e[0], t * v.e[1], t * v.e[2]);
}
  
inline Vec3 operator*(const Vec3 &v, double t){
// return Vec3(t * v.e[0], t * v.e[1], t * v.e[2]);
return t * v;//这里会直接复用上一个函数的逻辑，因此修改时只需要修改一处即可
}
  
using Point3 = Vec3; // 3D 空间中的点
using Color = Vec3; // RGB 颜色
```
